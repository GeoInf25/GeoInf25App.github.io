from js import document from pyodide.ffi import create_proxy def run_code(event): code_area = document.getElementById("code-area") output_div = document.getElementById("output") try: exec(code_area.value) except Exception as e: output_div.innerHTML = f"Error: {str(e)}" run_code_proxy = create_proxy(run_code) run_button = document.getElementById("run-button") run_button.addEventListener("click", run_code_proxy)
